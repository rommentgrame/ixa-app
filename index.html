import json import os import base64 from flask import Flask, request, jsonify, render_template_string app = Flask(__name__) # --- CONFIGURATION AUTOMATIQUE --- def load_config(): paths = ['config (1).json', 'config.json'] for path in paths: if os.path.exists(path): with open(path, 'r', encoding='utf-8') as f: return json.load(f) return {} config = load_config() API_KEY = config.get("groq_api_key", "") MODEL = config.get("groq_model", "llama-3.3-70b-versatile") # --- INTERFACE HTML (Intégrée) --- HTML_TEMPLATE = """ IXA v3 — Système Intégré
Historique chargé...
IXA CORE ONLINE
Assistant Vocal Direct
IXA v3

Prêt à vous écouter

""" # --- ROUTES SERVEUR --- @app.route('/') def home(): return render_template_string(HTML_TEMPLATE) @app.route('/api/chat', methods=['POST']) def chat_api(): data = request.json user_msg = data.get("message", "") files = data.get("files", []) # Intégration des fichiers dans le prompt full_prompt = user_msg if files: full_prompt += "\\n\\nCONCEPTS DOCUMENTS FOURNIS :\\n" for f in files: full_prompt += f"--- Fichier: {f['name']} ---\\n{f['content']}\\n" # Appel Groq direct depuis le serveur (Clé masquée pour l'utilisateur) import requests url = "https://api.groq.com/openai/v1/chat/completions" headers = {"Authorization": f"Bearer {API_KEY}", "Content-Type": "application/json"} payload = { "model": MODEL, "messages": [ {"role": "system", "content": "Tu es IXA. Analyse les documents si présents. Réponds avec précision."}, {"role": "user", "content": full_prompt} ] } res = requests.post(url, headers=headers, json=payload) return jsonify(res.json()) if __name__ == '__main__': print(f"Lancement de IXA v3 sur http://127.0.0.1:5000") print(f"Clé API chargée : {'OK' if API_KEY else 'NON TROUVÉE'}") app.run(debug=False, port=5000)
